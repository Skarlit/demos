Batch DOM update VS Individual DOM update

A test on DOM reflow + repaint speed by add 'target' class to 1000 list nodes

  
  .target {
    background: black;
    color: white;
    margin-left: 200px;
    padding: 10px;
    display: inline-block;
    margin: 12px;
    box-shadow: 0px 0px 4px 4px black;
    font-size: 12px;
  }

Batch update uses requestAnimation to repaint + reflow the DOM only once

        
  function update() {
    lis = document.getElementsByTagName('li');
    for(var i = 0; i < lis.length; i++) {
      lis[i].classList.add('target');
    }
    return true;
  }

  function nonBatchUpdate() {
    var promise = new Promise(function(resolve, reject) {
      var start = window.performance.now();
      if (update()) {
        resolve();  
      }
    });
    promise.then(function() {
      setTimeout(reset, 1000);
    });
  }

Sample Result:
Recalculate Style: 50.516ms;
Layout: 13.258ms;
Update Layer Tree: 0.052ms;
Paint: 4.237ms;
Composite Layers: 0.095ms;
Paint x 25: 8.892ms;
Total: 77.05

        
  function update() {
    lis = document.getElementsByTagName('li');
    for(var i = 0; i < lis.length; i++) {
      lis[i].classList.add('target');
    }
    return true;
  }

  function batchUpdate() {
    var promise = new Promise(function(resolve, reject) {
      window.requestAnimationFrame(update); 
      resolve(); 
    });
    promise.then(function() {
      setTimeout(reset, 1000);
    });
  }

Sample Result:
Animation Framed Fired(1)
Recalculate Style: 49.347ms
Layout: 9.515ms
Update Layer Tree: 0.042ms
Paint: 0.088ms
Composite Layers: 0.076ms
Paint x 26: 10.395ms
Total: 69.463ms

Conclusion: They are about the same performance on Chrome.
